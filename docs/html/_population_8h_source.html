274  Chromosome<T> *perform_crossover(Chromosome<T> *first, Chromosome<T> *second, RandomNumberGenerator *rnd) const {
275  auto *v = new std::vector<T *>(this->chromosomes->at(0)->get_genes()->size());
276 
277  int split_index;
278 
279  switch (this->configuration->get_crossover_type()) {
280  case (SINGLE_POINT_CROSSOVER): {
281  auto conf = (SinglePointCrossoverConfiguration *) this->configuration->get_crossover_configuration();
282 
283  switch (conf->get_mode()) {
284  case (RANDOM):
285  split_index = rnd->get_next(0, this->chromosomes->at(0)->get_genes()->size() - 1);
286 
287  break;
288  case (FIXED): {
289  auto fixed_conf = (FixedMode *) conf->get_single_point_crossover_configuration();
290 
291  split_index =
292  fixed_conf->get_crossover_factor() * this->chromosomes->at(0)->get_genes()->size();
293 
294  break;
295  }
296  }
297 
298  for (int i = 0; i < split_index; i++) {
299  v->at(i) = first->get_genes()->at(i);
300  }
301 
302  for (int i = split_index; i < this->chromosomes->at(0)->get_genes()->size(); i++) {
303  v->at(i) = second->get_genes()->at(i);
304  }
305 
306  break;
307  }
308  case (UNIFORM_CROSSOVER): {
309  for (int i = 0; i < this->chromosomes->at(0)->get_genes()->size(); i += 2) {
310  v->at(i) = first->get_genes()->at(i);
311  }
312 
313  for (int i = 1; i < this->chromosomes->at(0)->get_genes()->size(); i += 2) {
314  v->at(i) = first->get_genes()->at(i);
315  }
316 
317  break;
318  }
319  }
320 
321  auto mutation = rnd->get_next(0, 99) <
322  this->configuration->get_mutation_configuration()->get_mutation_rate() * 100;
323 
324  if (mutation) {
325  auto mutation_index = rnd->get_next(0, v->size() - 1);
326 
327  v->at(mutation_index) = this->configuration->get_alleles()->at(
328  rnd->get_next(0, this->configuration->get_alleles()->size() - 1));
329  }
330 
331 
332  auto *result = new Chromosome<T>(
333  const_cast<std::vector<T *> * > (v), this->configuration->fitness_function,
334  this->configuration->to_string);
335 
336  return result;

379  Chromosome<T> *perform_proportionate_selection(RandomNumberGenerator *rnd) const {
380  auto results = this->current_scores;
381  auto scores = new std::vector<double>();
382 
383  std::for_each(results->begin(), results->end(),
384  [&scores](const std::pair<Chromosome<T> *, double> *p) { scores->push_back(p->second); });
385 
386  double sum = std::accumulate(scores->begin(), scores->end(), 0.0);
387 
388  delete scores;
389 
390  auto rnd_share = rnd->get_next(0.0, sum);
391 
392  for (int i = 0; i < this->chromosomes->size(); i++) {
393  if (this->chromosomes->at(i)->get_fitness() > rnd_share) {
394  return this->chromosomes->at(i);
395  }
396 
397  rnd_share -= this->chromosomes->at(i)->get_fitness();
398  }

410  this->configuration->get_selection_configuration();
411

415  tournament_selection_configuration->get_n_contestants(), -1);
416 
417  for (int j = 0; j < tournament_selection_configuration->get_n_contestants(); j++) {
418  int rnd_index;
419 
420  do {
421  rnd_index = rnd->get_next(0, this->chromosomes->size() - 1);
422  } while (std::find(contestant_indexes->begin(), contestant_indexes->end(), rnd_index) !=
423  contestant_indexes->end());
424 
425  contestant_indexes->at(j) = rnd_index;
426  contestants->at(j) = this->chromosomes->at(rnd_index);
427  }
428 
429  auto contestants_results = this->get_scores(contestants);
430 
431  auto result = std::max_element(contestants_results->begin(), contestants_results->end(),
432  [](const std::pair<Chromosome<T> *, double> *l,
433  const std::pair<Chromosome<T> *, double> *r) {
434  return l->second < r->second;
435  });
436 
437  Chromosome<T> *r = contestants->at(std::distance(contestants_results->begin(), result));
438 
439  delete contestants;

443  [](const std::pair<Chromosome<T> *, double> *n) { delete n; });
444
